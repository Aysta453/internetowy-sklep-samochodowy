--  DDL for Function INCOMELEVEL
--------------------------------------------------------

  CREATE OR REPLACE EDITIONABLE FUNCTION "KRZYSIEK"."INCOMELEVEL" 
   ( name_in IN number )
   RETURN varchar2
IS
   monthly_value number(6);
   ILevel varchar2(20);

   cursor c1 is
     SELECT cena
     FROM pojazdy
     WHERE id_modelu = name_in;

BEGIN

   open c1;
   fetch c1 into monthly_value;
   close c1;

   IF monthly_value <= 40000 THEN
      ILevel := 'Low Income';

   ELSIF monthly_value > 40000 and monthly_value <= 70000 THEN
      ILevel := 'Avg Income';

   ELSIF monthly_value > 70000 and monthly_value <= 150000 THEN
      ILevel := 'SELECT * from marki;';

   ELSE
      ILevel := 'High Income';

   END IF;

   RETURN ILevel;

END IncomeLevel;
Commit;
